This example creates two instances of WireIt.Terminal into each block, and make the block draggable with WireIt.util.DD

		
		<style>
		div.blockBox {
			/* WireIt */
			position: absolute;
			z-index: 5;
			opacity: 0.8;

			/* Others */
			width: 100px;
			height: 100px;
			background-color: rgb(255,200,200);
			cursor: move;
		}
		</style>
		<script>
		window.onload = function() {

			var block1 = WireIt.cn('div', {className: "blockBox"}, {left: "478px", top: "75px"}, 'Move Me !');
			document.body.appendChild(block1);
			var terminals = [new WireIt.Terminal(block1), new WireIt.Terminal(block1)];
			new WireIt.util.DD(terminals,block1); 

			var block2 = WireIt.cn('div', {className: "blockBox"}, {left: "500px", top: "200px"}, 'Move Me !');
			document.body.appendChild(block2);
			var terminals = [new WireIt.Terminal(block2), new WireIt.Terminal(block2)];
			new WireIt.util.DD(terminals,block2);

		};
		</script>
